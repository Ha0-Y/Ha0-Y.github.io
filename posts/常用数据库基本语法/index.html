简单学习增删改查

  • 三个最广泛使用的开源 RDBMS：SQLite、MySQL 和 PostgreSQL。关系型数据库

  • 因为SQL的标准，三个数据库差不多的语法。

SQL

MySQL

其实是开源 MariaDB

数据库

-- 创建
CREATE DATABASE db_name; 

-- 删除
DROP DATABASE db_name;   

-- 查看
show databases;

-- 连接
use database_name;

表

-- 创建
CREATE TABLE table_name (
	column_name data_type 属性 
);

SHOW tables;

-- 删除
DROP TABLE table_name;

-- 查询
SELECT column_name from table_name;

-- 增加
INSERT into table_name(column_name) values("xxx");

查询限定

-- 排序
order by

-- 属性限定
WHERE, AND, OR

-- 结果个数

SQLite

sqlite3

命令

.help;     -- 帮助信息

.open test.db;  -- 打开文件，如果没有就创建

.show;     -- 显示一系列属性

.quit   -- 退出

.databases  -- 显示数据库

创建数据库

-- 读取或者创建
sqlite3 name
.open name

导入导出数据库

.dump   -- 将数据库导出SQL 文件 
sqlite3 test.db .dump > test.sql
sqlite3 test.db < test.db

创建表

.tables  -- 显示表
.schema  -- 表的信息

CREATE TABLE table_name (
	column_name type primary key
